\]

  \right\} \hskip.2in 
\]

\[
  V_l(e_t^\ast,m_t^\ast)=\left\{A(m_t^\ast)F^\alpha\cdot (l^\ast_t)^{1-\alpha}\left[P^W+\tau(\gamma(e_{t-1}^\ast))\right]-l_t^\ast-\mu_t^\ast-e_t^\ast+\beta V_l(e^\ast_{t+1},m^\ast_{t+1})
  \right\} \hskip.2in
